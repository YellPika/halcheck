105 template<
106 typename I,
107 typename... Args,
108 HALCHECK_REQUIRE(lib::is_input_iterator<I>()),
109 HALCHECK_REQUIRE(std::is_convertible<lib::iter_reference_t<I>, const_iterator>()),
110 HALCHECK_REQUIRE(std::is_constructible<T, Args...>())>
111 iterator emplace(I first, I last, Args &&...args) {
112 auto index = size();
113 _labels.emplace_back(std::forward<Args>(args)...);
114 _edges.emplace_back();
115 auto &parents = _edges.back().parents;
116
117 try {
118 if (first == last)
119 _roots.push_back(index);
120
121 while (first != last) {
122 const_iterator i = *first++;
123 if (_edges[i.index()].children.empty() || _edges[i.index()].children.back() != index)
124 _edges[i.index()].children.push_back(index);
125 parents.push_back(i.index());
126 }
127 } catch (...) {
128 for (auto &&edge : _edges) {
129 if (edge.children.back() == index)
130 edge.children.pop_back();
131 }
132
133 if (_roots.back() == index)
134 _roots.pop_back();
135
136 _labels.pop_back();
137 _edges.pop_back();
138 throw;
139 }
140
141 std::sort(parents.begin(), parents.end());
142 parents.erase(std::unique(parents.begin(), parents.end()), parents.end());
143
144 return iterator(_labels, index);
145 }

284 HALCHECK_REQUIRE(std::is_void<lib::invoke_result_t<F, lib::iterator_t<lib::dag<T>>>>())>
285void async(lib::dag<T> &graph, F func) {
286 std::vector<std::shared_future<void>> futures;
287 futures.reserve(graph.size());
288
289 for (auto i = graph.begin(); i != graph.end(); ++i) {
290 std::vector<std::shared_future<void>> parents;
291 for (auto j : graph.parents(i))
292 parents.push_back(futures[j - graph.begin()]);
293
294 futures.emplace_back(std::async(
295 std::launch::async,
296 [&func, i](const std::vector<std::shared_future<void>> &parents) {
297 for (const auto &parent : parents)
298 parent.wait();
299
300 lib::invoke(func, i);
301 },
302 std::move(parents)));
303 }
304
305 for (auto &&future : futures)
306 future.get();
307}

338 HALCHECK_REQUIRE(!std::is_void<lib::invoke_result_t<F, lib::iterator_t<lib::dag<T>>>>())>
339lib::dag<lib::invoke_result_t<F, lib::iterator_t<lib::dag<T>>>> async(lib::dag<T> &graph, F func) {
340 using result = lib::invoke_result_t<F, lib::iterator_t<lib::dag<T>>>;
341
342 std::vector<std::promise<result>> results(graph.size());
343 lib::async(graph, [&](lib::iterator_t<lib::dag<T>> it) {
344 results[it - graph.begin()].set_value(lib::invoke(func, it));
345 });
346
347 lib::dag<result> output;
348 output.reserve(graph.size());
349
350 for (std::size_t i = 0; i < results.size(); i++) {
351 output.emplace(
352 lib::transform(
353 graph.parents(graph.begin() + i),
354 [&](lib::iterator_t<lib::dag<T>> it) { return output.begin() + (it - graph.begin()); }),
355 results[i].get_future().get());

381template<typename T, typename F>
382bool linearize(
383 const lib::dag<F> &dag,
384 T &seed,
385 std::vector<std::size_t> &references,
386 std::vector<typename lib::dag<F>::const_iterator> &queue) {
387 for (std::size_t i = 0; i < queue.size(); ++i) {
388 auto next = seed;
389 auto it = queue[i];
390 if (!lib::invoke(*it, next))
391 continue;
392
393 for (auto j : dag.children(it)) {
394 if (--references[j - dag.begin()] == 0)
395 queue.push_back(j);
396 }
397
398 std::swap(queue[i], queue.back());
399 queue.pop_back();
400
401 auto _ = lib::finally([&] {
402 queue.push_back(it);
403 std::swap(queue[i], queue.back());
404
405 for (auto j : dag.children(it)) {
406 if (references[j - dag.begin()]++ == 0)
407 queue.pop_back();
408 }
409 });
410
411 if (linearize(dag, next, references, queue)) {
412 seed = std::move(next);
413 return true;
414 }
415 }
416
417 return queue.empty();
418}

424 HALCHECK_REQUIRE(lib::is_copyable<T>()),

426bool linearize(const lib::dag<F> &dag, T &seed) {
427 std::vector<std::size_t> references(dag.size(), 0);
428 for (auto i = dag.begin(); i != dag.end(); ++i) {
429 for (auto j : dag.children(i))
